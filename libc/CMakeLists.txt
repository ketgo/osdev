CMAKE_MINIMUM_REQUIRED(VERSION 2.8.0)

include(ExternalProject)
ExternalProject_Add(
    libk
    SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}
    CONFIGURE_COMMAND ${CMAKE_CURRENT_SOURCE_DIR}/configure --prefix=/usr --host=${CMAKE_SYSTEM_PROCESSOR}-elf --target=${CMAKE_SYSTEM_PROCESSOR}
    BUILD_COMMAND make
    INSTALL_COMMAND make DESTDIR=${CMAKE_CURRENT_SOURCE_DIR}/../build/${CMAKE_SYSTEM_PROCESSOR}/sysroot install
)
